Function Properties
Language: PLPGSQL
Return Type: integer

registry_set_int4(key, value) Set or delete a registry value 

DECLARE
	p_key		alias for $1;
	p_value		alias for $2;
BEGIN
	if p_value is null then
		delete from sl_registry
				where reg_key = p_key;
	else
		lock table sl_registry;
		update sl_registry
				set reg_int4 = p_value
				where reg_key = p_key;
		if not found then
			insert into sl_registry (reg_key, reg_int4)
					values (p_key, p_value);
		end if;
	end if;
	return p_value;
END;
